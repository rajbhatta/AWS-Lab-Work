#  Load .sql file located inside /home/ubuntu/test.SQL #
```js
sudo mysqldump \
    --databases rdsdb \
    --master-data=2  \
    --single-transaction \
    --order-by-primary \
    -r /home/ubunt/test.sql \
    -u admin \
    -p admin 
```
